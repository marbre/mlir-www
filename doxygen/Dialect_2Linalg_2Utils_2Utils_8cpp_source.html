47 // Helper visitor to determine whether an AffineExpr is tiled.
48 // This is achieved by traversing every AffineDimExpr with position `pos` and
49 // checking whether the corresponding `tileSizes[pos]` is non-zero.
50 // This also enforces only positive coefficients occur in multiplications.
51 //
52 // Example:
53 // `d0 + 2 * d1 + d3` is tiled by [0, 0, 0, 2] but not by [0, 0, 2, 0]
54 //
55 struct TileCheck : public AffineExprVisitor<TileCheck> {
56  TileCheck(ValueRange tileSizes) : isTiled(false), tileSizes(tileSizes) {}
57 
58  void visitDimExpr(AffineDimExpr expr) {
59  isTiled |= !isZero(tileSizes[expr.getPosition()]);
60  }
61  void visitAffineBinaryOpExpr(AffineBinaryOpExpr expr) {
62  visit(expr.getLHS());
63  visit(expr.getRHS());
64  if (expr.getKind() == mlir::AffineExprKind::Mul)
65  assert(expr.getRHS().cast<AffineConstantExpr>().getValue() > 0 &&
66  "nonpositive multiplying coefficient");
67  }
68  bool isTiled;
69  ValueRange tileSizes;
70 };

92 Optional<RegionMatcher::BinaryOpKind>
93 RegionMatcher::matchAsScalarBinaryOp(GenericOp op) {
94  auto &region = op.region();
95  if (!llvm::hasSingleElement(region))
96  return llvm::None;
97 
98  Block &block = region.front();
99  if (block.getNumArguments() != 2 ||
100  !block.getArgument(0).getType().isSignlessIntOrFloat() ||
101  !block.getArgument(1).getType().isSignlessIntOrFloat())
102  return llvm::None;
103 
104  auto &ops = block.getOperations();
105  if (!llvm::hasSingleElement(block.without_terminator()))
106  return llvm::None;
107 
108  using mlir::matchers::m_Val;
109  auto a = m_Val(block.getArgument(0));
110  auto b = m_Val(block.getArgument(1));
111 
112  auto addPattern = m_Op<linalg::YieldOp>(m_Op<AddIOp>(a, b));
113  if (addPattern.match(&ops.back()))
114  return BinaryOpKind::IAdd;
115 
116  return llvm::None;
117 }

183 /// If `size` comes from an AffineMinOp and one of the values of AffineMinOp
184 /// is a constant then return a new value set to the smallest such constant.
185 /// Otherwise returngetSmallestBoundingIndex nullptr.
186 IntegerAttr getSmallestBoundingIndex(Value size) {
187  Optional<int64_t> boundingConst = {};
188  if (auto affineMinOp = size.getDefiningOp<AffineMinOp>()) {
189  for (auto e : affineMinOp.getAffineMap().getResults())
190  if (auto cst = e.dyn_cast<AffineConstantExpr>())
191  boundingConst = boundingConst
192  ? std::min(boundingConst.getValue(), cst.getValue())
193  : cst.getValue();
194  } else if (auto constIndexOp = size.getDefiningOp<ConstantOp>()) {
195  if (constIndexOp.getType().isa<IndexType>())
196  boundingConst = constIndexOp.value().cast<IntegerAttr>().getInt();
197  } else if (auto affineApplyOp = size.getDefiningOp<AffineApplyOp>()) {
198  if (auto cExpr = affineApplyOp.getAffineMap()
199  .getResult(0)
200  .dyn_cast<AffineConstantExpr>())
201  boundingConst = cExpr.getValue();
202  } else if (auto dimOp = size.getDefiningOp<tensor::DimOp>()) {
203  auto shape = dimOp.source().getType().dyn_cast<ShapedType>();
204  if (auto constOp = dimOp.index().getDefiningOp<ConstantOp>()) {
205  if (auto indexAttr = constOp.value().dyn_cast<IntegerAttr>()) {
206  auto dimIndex = indexAttr.getInt();
207  if (!shape.isDynamicDim(dimIndex)) {
208  boundingConst = shape.getShape()[dimIndex];
209  }
210  }
211  }
212  }
213  if (boundingConst && *boundingConst >= 0)
214  return Builder(size.getContext()).getIndexAttr(*boundingConst);
215  return nullptr;
216 }

218 /// Specialization to build an scf "for" nest.
219 template <>
220 void GenerateLoopNest<scf::ForOp>::doit(
221  OpBuilder &b, Location loc, ArrayRef<Range> loopRanges, LinalgOp linalgOp,
222  ArrayRef<Attribute> iteratorTypes,
223  function_ref<scf::ValueVector(OpBuilder &, Location, ValueRange,
224  ValueRange)>
225  bodyBuilderFn,
226  Optional<LinalgLoopDistributionOptions> distributionOptions,
227  ArrayRef<StringRef> distributionTypes) {
228  SmallVector<Value> iterArgInitValues = linalgOp.getOutputTensorOperands();
229  // Create procInfo so it dominates loops, if appropriate.
230  SmallVector<ProcInfo, 4> procInfo;
231  SmallVector<DistributionMethod, 0> distributionMethod;
232  if (distributionOptions.hasValue()) {
233  // Collect loop ranges for parallel dimensions.
234  SmallVector<Range, 2> parallelLoopRanges;
235  for (auto iteratorType : enumerate(iteratorTypes))
236  if (isParallelIteratorType(iteratorType.value()))
237  parallelLoopRanges.push_back(loopRanges[iteratorType.index()]);
238 
239  // Get their distribution schemes.
240  distributionMethod = distributionOptions->distributionMethod;
241  if (distributionMethod.size() < parallelLoopRanges.size())
242  parallelLoopRanges.resize(distributionMethod.size());
243  procInfo = distributionOptions->procInfo(b, loc, parallelLoopRanges);
244  }
245 
246  SmallVector<Value, 4> lbs, ubs, steps;
247  unpackRanges(loopRanges, lbs, ubs, steps);
248  LoopNest loopNest = mlir::scf::buildLoopNest(
249  b, loc, lbs, ubs, steps, iterArgInitValues, bodyBuilderFn);
250 
251  if (!distributionOptions || loopNest.loops.empty())
252  return;
253 
254  // Filter out scf.for loops that were created out of parallel dimensions.
255  SmallVector<scf::ForOp, 4> loops;
256  for (auto iteratorType : enumerate(iteratorTypes))
257  if (isParallelIteratorType(iteratorType.value()))
258  loops.push_back(loopNest.loops[iteratorType.index()]);
259 
260  // Distribute - only supports cyclic distribution for now.
261  for (auto it : llvm::zip(loops, procInfo, distributionMethod))
262  if (std::get<2>(it) == DistributionMethod::Cyclic)
263  mapLoopToProcessorIds(std::get<0>(it), std::get<1>(it).procId,
264  std::get<1>(it).nprocs);
265 }

297 template <>
298 void GenerateLoopNest<TiledLoopOp>::doit(
299  OpBuilder &b, Location loc, ArrayRef<Range> loopRanges, LinalgOp linalgOp,
300  ArrayRef<Attribute> iteratorTypes,
301  function_ref<scf::ValueVector(OpBuilder &, Location, ValueRange,
302  ValueRange)>
303  bodyBuilderFn,
304  Optional<LinalgLoopDistributionOptions> distributionOptions,
305  ArrayRef<StringRef> distributionTypes) {
306  SmallVector<ProcInfo, 2> procInfo;
307  SmallVector<Value, 4> lbs, ubs, steps;
308  unpackRanges(loopRanges, lbs, ubs, steps);

310  auto wrappedBuilderFn = [&](OpBuilder &nestedBuilder, Location nestedLoc,
311  ValueRange ivs, ValueRange inputs,
312  ValueRange outputs) {
313  SmallVector<Value> outputTensors = linalgOp.getOutputTensorOperands();
314  scf::ValueVector results =
315  bodyBuilderFn(nestedBuilder, nestedLoc, ivs, outputTensors);
316  nestedBuilder.create<linalg::YieldOp>(nestedLoc, results);
317  };

320  SmallVector<Value> outputOperands = linalgOp.getOutputOperands();
321  auto tiledLoop =
322  b.create<TiledLoopOp>(loc, lbs, ubs, steps, inputOperands, outputOperands,
323  b.getArrayAttr(iteratorTypes), wrappedBuilderFn);
324  if (!distributionTypes.empty())
325  tiledLoop.setDistributionTypes(b, distributionTypes);
326

328  auto isInsideTiledLoop = [&](OpOperand &operand) {
329  return operand.getOwner()->getBlock() == tiledLoop.getBody();
330  };
331  for (auto it : llvm::zip(inputOperands, tiledLoop.getRegionInputArgs()))
332  std::get<0>(it).replaceUsesWithIf(std::get<1>(it), isInsideTiledLoop);
333  for (auto it : llvm::zip(outputOperands, tiledLoop.getRegionOutputArgs()))
334  std::get<0>(it).replaceUsesWithIf(std::get<1>(it), isInsideTiledLoop);
335 }

348 /// Generates a loop nest consisting of scf.parallel and scf.for, depending
349 /// on the `iteratorTypes.` Consecutive parallel loops create a single
350 /// scf.parallel operation; each sequential loop creates a new scf.for
351 /// operation. The body of the innermost loop is populated by
352 /// `bodyBuilderFn` that accepts a range of induction variables for all
353 /// loops. `ivStorage` is used to store the partial list of induction
354 /// variables.
355 // TODO: this function can be made iterative instead. However, it
356 // will have at most as many recursive calls as nested loops, which rarely
357 // exceeds 10.
358 static void generateParallelLoopNest(
359  OpBuilder &b, Location loc, ValueRange lbs, ValueRange ubs,
360  ValueRange steps, ArrayRef<Attribute> iteratorTypes,
361  function_ref<void(OpBuilder &, Location, ValueRange)> bodyBuilderFn,
362  SmallVectorImpl<Value> &ivStorage,
363  ArrayRef<DistributionMethod> distributionMethod = {}) {
364  assert(lbs.size() == ubs.size());
365  assert(lbs.size() == steps.size());
366  assert(lbs.size() == iteratorTypes.size());
367 
368  // If there are no (more) loops to be generated, generate the body and be
369  // done with it.
370  if (iteratorTypes.empty()) {
371  bodyBuilderFn(b, loc, ivStorage);
372  return;
373  }
374 
375  // Find the outermost parallel loops and drop their types from the list.
376  unsigned nLoops = iteratorTypes.size();
377  unsigned nOuterPar =
378  nLoops - iteratorTypes.drop_while(isParallelIteratorType).size();
379 
380  // If there are no outer parallel loops, generate one sequential loop and
381  // recurse. Note that we wouldn't have dropped anything from `iteratorTypes`
382  // in this case.
383  if (nOuterPar == 0) {
384  LoopNest singleLoop = buildLoopNest(
385  b, loc, lbs.take_front(), ubs.take_front(), steps.take_front(),
386  [&](OpBuilder &b, Location loc, ValueRange ivs) {
387  ivStorage.append(ivs.begin(), ivs.end());
388  generateParallelLoopNest(b, loc, lbs.drop_front(), ubs.drop_front(),
389  steps.drop_front(),
390  iteratorTypes.drop_front(), bodyBuilderFn,
391  ivStorage, distributionMethod);
392  });
393  return;
394  }
395  if (distributionMethod.empty()) {
396  // Generate a single parallel loop-nest operation for all outermost
397  // parallel loops and recurse.
398  b.create<scf::ParallelOp>(
399  loc, lbs.take_front(nOuterPar), ubs.take_front(nOuterPar),
400  steps.take_front(nOuterPar),
401  [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange localIvs) {
402  ivStorage.append(localIvs.begin(), localIvs.end());
403  generateParallelLoopNest(
404  nestedBuilder, nestedLoc, lbs.drop_front(nOuterPar),
405  ubs.drop_front(nOuterPar), steps.drop_front(nOuterPar),
406  iteratorTypes.drop_front(nOuterPar), bodyBuilderFn, ivStorage,
407  (distributionMethod.size() < nOuterPar)
408  ? ArrayRef<DistributionMethod>()
409  : distributionMethod.drop_front(nOuterPar));
410  });
411  return;
412  }
413 
414  // Process all consecutive similarly distributed loops simultaneously.
415  DistributionMethod methodToUse = distributionMethod[0];
416  unsigned numProcessed = 1;
417  for (unsigned i = 1; i < nOuterPar && i < distributionMethod.size(); ++i) {
418  if (distributionMethod[i] != methodToUse)
419  break;
420  numProcessed++;
421  }
422 
423  switch (methodToUse) {
424  case DistributionMethod::Cyclic: {
425  // Generate a single parallel loop-nest operation for all outermost
426  // parallel loops and recurse.
427  b.create<scf::ParallelOp>(
428  loc, lbs.take_front(numProcessed), ubs.take_front(numProcessed),
429  steps.take_front(numProcessed),
430  [&](OpBuilder &nestedBuilder, Location nestedLoc, ValueRange localIvs) {
431  ivStorage.append(localIvs.begin(), localIvs.end());
432  generateParallelLoopNest(
433  nestedBuilder, nestedLoc, lbs.drop_front(numProcessed),
434  ubs.drop_front(numProcessed), steps.drop_front(numProcessed),
435  iteratorTypes.drop_front(numProcessed), bodyBuilderFn, ivStorage,
436  (distributionMethod.size() < numProcessed)
437  ? ArrayRef<DistributionMethod>()
438  : distributionMethod.drop_front(numProcessed));
439  });
440  return;
441  }
442  case DistributionMethod::CyclicNumProcsGeNumIters: {
443  // Check (for the processed loops) that the iteration is in-bounds.
444  ArithBuilder ab(b, loc);
445  Value cond = ab.slt(lbs[0], ubs[0]);
446  for (unsigned i = 1; i < numProcessed; ++i)
447  cond = ab._and(cond, ab.slt(lbs[i], ubs[i]));
448  ivStorage.append(lbs.begin(), std::next(lbs.begin(), numProcessed));
449  b.create<scf::IfOp>(loc, cond, [&](OpBuilder &b, Location loc) {
450  generateParallelLoopNest(
451  b, loc, lbs.drop_front(numProcessed), ubs.drop_front(numProcessed),
452  steps.drop_front(numProcessed),
453  iteratorTypes.drop_front(numProcessed), bodyBuilderFn, ivStorage,
454  distributionMethod.drop_front(numProcessed));
455  b.create<scf::YieldOp>(loc, ValueRange{});
456  });
457  return;
458  }
459  case DistributionMethod::CyclicNumProcsEqNumIters:
460  // No check/loops needed here. Set the `%iv` to be the `%lb` and proceed
461  // with inner loop generation.
462  ivStorage.append(lbs.begin(), std::next(lbs.begin(), numProcessed));
463  generateParallelLoopNest(
464  b, loc, lbs.drop_front(numProcessed), ubs.drop_front(numProcessed),
465  steps.drop_front(numProcessed), iteratorTypes.drop_front(numProcessed),
466  bodyBuilderFn, ivStorage, distributionMethod.drop_front(numProcessed));
467  return;
468  }
469 }

471 /// Specialization for generating a mix of parallel and sequential scf loops.
472 template <>
473 void GenerateLoopNest<scf::ParallelOp>::doit(
474  OpBuilder &b, Location loc, ArrayRef<Range> loopRanges, LinalgOp linalgOp,
475  ArrayRef<Attribute> iteratorTypes,
476  function_ref<scf::ValueVector(OpBuilder &, Location, ValueRange,
477  ValueRange)>
478  bodyBuilderFn,
479  Optional<LinalgLoopDistributionOptions> distributionOptions,
480  ArrayRef<StringRef> distributionTypes) {
481  SmallVector<Value> iterArgInitValues = linalgOp.getOutputTensorOperands();
482  assert(iterArgInitValues.empty() && "unexpected ParallelOp init values");
483  // This function may be passed more iterator types than ranges.
484  assert(iteratorTypes.size() >= loopRanges.size() &&
485  "expected iterator type for all ranges");
486  iteratorTypes = iteratorTypes.take_front(loopRanges.size());
487  SmallVector<Value, 8> lbsStorage, ubsStorage, stepsStorage, ivs;
488  unsigned numLoops = iteratorTypes.size();
489  ivs.reserve(numLoops);
490  lbsStorage.reserve(numLoops);
491  ubsStorage.reserve(numLoops);
492  stepsStorage.reserve(numLoops);
493 
494  // Get the loop lb, ub, and step.
495  unpackRanges(loopRanges, lbsStorage, ubsStorage, stepsStorage);
496 
497  // Modify the lb, ub, and step based on the distribution options.
498  SmallVector<DistributionMethod, 0> distributionMethod;
499  if (distributionOptions) {
500  auto &options = distributionOptions.getValue();
501  distributionMethod.assign(distributionOptions->distributionMethod.begin(),
502  distributionOptions->distributionMethod.end());
503  SmallVector<Range, 2> parallelLoopRanges;
504  for (auto iteratorType : enumerate(iteratorTypes)) {
505  if (isParallelIteratorType(iteratorType.value()))
506  parallelLoopRanges.push_back(loopRanges[iteratorType.index()]);
507  }
508  if (distributionMethod.size() < parallelLoopRanges.size())
509  parallelLoopRanges.resize(distributionMethod.size());
510  SmallVector<ProcInfo, 2> procInfo =
511  options.procInfo(b, loc, parallelLoopRanges);
512  unsigned index = 0;
513  for (auto iteratorType : enumerate(iteratorTypes)) {
514  if (index >= procInfo.size())
515  break;
516  if (isParallelIteratorType(iteratorType.value())) {
517  unsigned i = iteratorType.index();
518  updateBoundsForCyclicDistribution(b, loc, procInfo[index].procId,
519  procInfo[index].nprocs, lbsStorage[i],
520  ubsStorage[i], stepsStorage[i]);
521  index++;
522  }
523  }
524  }
525  ValueRange lbs(lbsStorage), ubs(ubsStorage), steps(stepsStorage);
526  generateParallelLoopNest(
527  b, loc, lbs, ubs, steps, iteratorTypes,
528  [&](OpBuilder &b, Location loc, ValueRange ivs) {
529  bodyBuilderFn(b, loc, ivs, {});
530  },
531  ivs, distributionMethod);
532 
533  assert(ivs.size() == iteratorTypes.size() && "did not generate enough loops");
534 }

536 Value makeTiledShape(OpBuilder &builder, Location loc, Value valueToTile,
537  ValueRange tileSizes, AffineMap map, ValueRange lbs,
538  ValueRange subShapeSizes) {
539  auto shapedType = valueToTile.getType().dyn_cast<ShapedType>();
540  assert(shapedType && "only shaped types can be tiled");
541  ArrayRef<int64_t> shape = shapedType.getShape();
542  int64_t rank = shapedType.getRank();
543 
544  // Construct a new subview / extract_slice for the tile.
545  SmallVector<OpFoldResult, 4> offsets, sizes, strides;
546  offsets.reserve(rank);
547  sizes.reserve(rank);
548  strides.reserve(rank);
549  for (unsigned r = 0; r < rank; ++r) {
550  LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: for dim#" << r);
551  if (!isTiled(map.getSubMap({r}), tileSizes)) {
552  offsets.push_back(builder.getIndexAttr(0));
553  Value dim = createOrFoldDimOp(builder, loc, valueToTile, r);
554  sizes.push_back(getAsOpFoldResult(dim));
555  strides.push_back(builder.getIndexAttr(1));
556  LLVM_DEBUG(llvm::dbgs() << ": not tiled: use size: " << dim << "\n");
557  continue;
558  }
559  LLVM_DEBUG(llvm::dbgs() << ": tiled: figure out subsize...\n");
560 
561  // Tiling creates a new slice at the proper index, the slice step is 1
562  // (i.e. the op does not subsample, stepping occurs in the loop).
563  auto m = map.getSubMap({r});
564  LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: submap: " << m << "\n");
565  auto offset = applyMapToValues(builder, loc, m, lbs).front();
566  offsets.push_back(offset);
567  auto closedIntSize =
568  applyMapToValues(builder, loc, m, subShapeSizes).front();
569  // Resulting size needs to be made half open interval again.
570  AffineExpr s0 = getAffineSymbolExpr(0, builder.getContext());
571  Value size = makeComposedAffineApply(builder, loc, s0 + 1, closedIntSize);
572  LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: raw size: " << size << "\n");
573 
574  // The size of the subview / extract_slice should be trimmed to avoid
575  // out-of-bounds accesses, unless we statically know the subshape size
576  // divides the shape size evenly.
577  int64_t shapeSize = shape[r];
578  auto sizeCst = size.getDefiningOp<ConstantIndexOp>();
579  if (ShapedType::isDynamic(shapeSize) || !sizeCst ||
580  (shapeSize % sizeCst.getValue()) != 0) {
581  LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: shapeSize=" << shapeSize
582  << ", size: " << size
583  << ": make sure in bound with affine.min\n");
584  AffineExpr dim0, dim1, dim2;
585  bindDims(builder.getContext(), dim0, dim1, dim2);
586  // Compute min(size, dim - offset) to avoid out-of-bounds accesses.
587  AffineMap minMap =
588  AffineMap::inferFromExprList(
589  ArrayRef<ArrayRef<AffineExpr>>{{dim0, dim1 - dim2}})
590  .front();
591  Value d = createOrFoldDimOp(builder, loc, valueToTile, r);
592  SmallVector<Value, 4> operands{size, d, offset};
593  fullyComposeAffineMapAndOperands(&minMap, &operands);
594  size = builder.create<AffineMinOp>(loc, builder.getIndexType(), minMap,
595  operands);
596  }
597 
598  sizes.push_back(size);
599  LLVM_DEBUG(llvm::dbgs()
600  << "makeTiledShape: new offset: " << offset << "\n");
601  LLVM_DEBUG(llvm::dbgs() << "makeTiledShape: new size: " << size << "\n");
602  strides.push_back(builder.getIndexAttr(1));
603  }
604 
605  Operation *sliceOp = shapedType.isa<MemRefType>()
606  ? builder
607  .create<memref::SubViewOp>(
608  loc, valueToTile, offsets, sizes, strides)
609  .getOperation()
610  : builder
611  .create<tensor::ExtractSliceOp>(
612  loc, valueToTile, offsets, sizes, strides)
613  .getOperation();
614  return sliceOp->getResult(0);
615 }

617 SmallVector<Value> computeTileOffsets(OpBuilder &b, Location loc,
618  ValueRange ivs, ValueRange tileSizes) {
619  SmallVector<Value> offsets;
620  for (unsigned idx = 0, idxIvs = 0, e = tileSizes.size(); idx < e; ++idx) {
621  LLVM_DEBUG(llvm::dbgs() << "makeTiledShapes: for loop#" << idx << "\n");
622  bool isTiled = !isZero(tileSizes[idx]);
623  offsets.push_back(isTiled ? ivs[idxIvs++]
624  : b.create<ConstantIndexOp>(loc, 0).getResult());
625  LLVM_DEBUG(llvm::dbgs()
626  << "computeTileOffsets: " << offsets.back() << "\n");
627  }
628  return offsets;
629 }

631 SmallVector<Value> computeTileSizes(OpBuilder &b, Location loc, ValueRange ivs,
632  ValueRange tileSizes,
633  ArrayRef<Value> sizeBounds) {
634  SmallVector<Value> sizes;
635  for (unsigned idx = 0, e = tileSizes.size(); idx < e; ++idx) {
636  bool isTiled = !isZero(tileSizes[idx]);
637  // Before composing, we need to make range a closed interval.
638  Value size = isTiled ? tileSizes[idx] : sizeBounds[idx];
639  AffineExpr d0 = getAffineDimExpr(0, b.getContext());
640  sizes.push_back(makeComposedAffineApply(b, loc, d0 - 1, size));
641  LLVM_DEBUG(llvm::dbgs() << "computeTileSizes: " << sizes.back() << "\n");
642  }
643  return sizes;
644 }

646 SmallVector<Value, 4> makeTiledShapes(OpBuilder &b, Location loc,
647  LinalgOp linalgOp,
648  ArrayRef<Value> valuesToTile,
649  ValueRange ivs, ValueRange tileSizes,
650  ArrayRef<Value> sizeBounds) {
651  assert(ivs.size() == static_cast<size_t>(llvm::count_if(
652  llvm::make_range(tileSizes.begin(), tileSizes.end()),
653  [](Value v) { return !isZero(v); })) &&
654  "expected as many ivs as non-zero sizes");
655 
656  // Construct (potentially temporary) mins and maxes on which to apply maps
657  // that define tile subshapes.
658  SmallVector<Value> lbs = computeTileOffsets(b, loc, ivs, tileSizes);
659  SmallVector<Value> subShapeSizes =
660  computeTileSizes(b, loc, ivs, tileSizes, sizeBounds);
661 
662  assert(static_cast<int64_t>(valuesToTile.size()) ==
663  linalgOp.getNumInputsAndOutputs() &&
664  "expected one value to tile for every operand");
665  SmallVector<Value, 4> tiledShapes;
666  tiledShapes.reserve(valuesToTile.size());
667  for (OpOperand *opOperand : linalgOp.getInputAndOutputOperands()) {
668  Value shapedOp = valuesToTile[opOperand->getOperandNumber()];
669  LLVM_DEBUG(llvm::dbgs() << "makeTiledShapes: for operand " << shapedOp);
670  AffineMap map = linalgOp.getTiedIndexingMap(opOperand);
671  // If the shape is not tiled, we can use it as is.
672  if (!isTiled(map, tileSizes)) {
673  tiledShapes.push_back(shapedOp);
674  LLVM_DEBUG(llvm::dbgs() << ": not tiled: use shape: "
675  << opOperand->get().getType() << "\n");
676  continue;
677  }
678  LLVM_DEBUG(llvm::dbgs() << ": tiled: figure out subshape...\n");
679 
680  tiledShapes.push_back(
681  makeTiledShape(b, loc, shapedOp, tileSizes, map, lbs, subShapeSizes));
682  }
683 
684  return tiledShapes;
685 }
