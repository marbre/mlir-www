42 // Statically known information for a particular Value.
43 //
44 // This struct currently tracks only information relevant for tensor/array-like
45 // shaped types. It is fine to associate a `ValueKnowledge` with a non-shaped
46 // type as long as it is in the default "no knowledge" state returned by
47 // `getPessimisticValueState`. The important invariant is that we cannot
48 // claim to know something about a value which is false.
49 //
50 // This class could also be called "dataflow facts", "lattice value", etc.
51 struct ValueKnowledge {
52  ValueKnowledge() = delete;
53  ValueKnowledge(bool hasSizes, std::vector<int64_t> sizes, Type dtype)
54  : hasSizes(hasSizes), sizes(sizes), dtype(dtype) {
55  assert(sizes.size() == 0 || hasSizes);
56  }
57 
58  // Get the static knowledge intrinsic to `type`.
59  static ValueKnowledge getKnowledgeFromType(Type type) {
60  ValueKnowledge result = getPessimisticValueState(type.getContext());
61  if (auto shapedType = type.dyn_cast<ShapedType>()) {
62  if (shapedType.hasRank()) {
63  result.hasSizes = true;
64  result.sizes = shapedType.getShape();
65  }
66  result.dtype = shapedType.getElementType();
67  }
68  return result;
69  }
70 
71  // Return a pessimistic/conservative value state without assuming any knowlege
72  // about the IR.
73  static ValueKnowledge getPessimisticValueState(MLIRContext *context) {
74  return ValueKnowledge(false, {}, Type());
75  }
76 
77  Type getType() const {
78  if (hasSizes) {
79  return RankedTensorType::get(llvm::makeArrayRef(sizes), dtype);
80  }
81  return UnrankedTensorType::get(dtype);
82  }
83 
84  bool operator==(const ValueKnowledge &rhs) const {
85  return std::make_tuple(hasSizes, sizes, dtype) ==
86  std::make_tuple(rhs.hasSizes, rhs.sizes, rhs.dtype);
87  }
88 
89  // Given two pieces of static knowledge, calculate conservatively the
90  // information we can be sure about.
91  static ValueKnowledge join(const ValueKnowledge &lhs,
92  const ValueKnowledge &rhs) {
93  // Mental model: All conditions are checking how to change from the safe "no
94  // knowledge" default-initialized state to a state with more knowledge
95  // consistent with lhs and rhs.
96  ValueKnowledge result = getPessimisticValueState(nullptr);
97 
98  if (lhs.hasSizes && !rhs.hasSizes) {
99  result.hasSizes = true;
100  result.sizes = lhs.sizes;
101  } else if (!lhs.hasSizes && rhs.hasSizes) {
102  result.hasSizes = true;
103  result.sizes = rhs.sizes;
104  } else if (lhs.hasSizes && rhs.hasSizes &&
105  lhs.sizes.size() == rhs.sizes.size()) {
106  result.hasSizes = true;
107  result.sizes.resize(lhs.sizes.size(), ShapedType::kDynamicSize);
108  for (int i = 0, e = result.sizes.size(); i != e; i++) {
109  int64_t lhsSize = lhs.sizes[i];
110  int64_t rhsSize = rhs.sizes[i];
111  int64_t &resultSize = result.sizes[i];
112  if (lhsSize == ShapedType::kDynamicSize) {
113  resultSize = rhsSize;
114  } else if (rhsSize == ShapedType::kDynamicSize) {
115  resultSize = lhsSize;
116  } else if (lhsSize == rhsSize) {
117  resultSize = lhsSize;
118  }
119  }
120  }
121 
122  result.dtype = joinElementTypes(lhs.dtype, rhs.dtype);
123  return result;
124  }
125 
126  // Whether the Value is known to have a list of sizes.
127  bool hasSizes;
128  // If `hasSizes`, the sizes along each rank. Unknown sizes are represented as
129  // `ShapedType::kDynamicSize`.
130  std::vector<int64_t> sizes;
131  // The dtype of a tensor.
132  // This is equal to nullptr if we don't know that it is a specific concrete
133  // type.
134  Type dtype;
135 };

139 /// Pass that enables broadcast by making all input arrays have the same
140 /// number of dimensions. Insert RESHAPE operations to lower rank operand
141 struct TosaInferShapes : public TosaInferShapesBase<TosaInferShapes> {
142 public:
143  void runOnFunction() override {
144  FuncOp func = getOperation();
145 
146  IRRewriter rewriter(func.getContext());
147 
148  func.walk([&](Operation *op) {
149  if (op->getDialect()->getNamespace() !=
150  tosa::TosaDialect::getDialectNamespace())
151  return;
152  InferShapedTypeOpInterface shapeInterface =
153  dyn_cast<InferShapedTypeOpInterface>(op);
154  if (!shapeInterface)
155  return;
156 
157  SmallVector<ShapedTypeComponents> returnedShapes;
158  if (shapeInterface
159  .inferReturnTypeComponents(
160  op->getContext(), op->getLoc(), op->getOperands(),
161  op->getAttrDictionary(), op->getRegions(), returnedShapes)
162  .succeeded()) {
163  for (auto it : llvm::zip(op->getResults(), returnedShapes)) {
164  Value result = std::get<0>(it);
165  ShapedTypeComponents predictedShape = std::get<1>(it);
166 
167  // Check whether this use case is replaceable. We define an op as
168  // being replaceable if it is used by a ReturnOp or a TosaOp.
169  bool replaceable = true;
170  for (auto user : result.getUsers()) {
171  if (isa<ReturnOp>(user))
172  continue;
173  if (user->getDialect()->getNamespace() ==
174  tosa::TosaDialect::getDialectNamespace())
175  continue;
176 
177  replaceable = false;
178  }
179 
180  // Determine the knowledge based on the output type.
181  Type resultTy = result.getType();
182  auto currentKnowledge =
183  ValueKnowledge::getKnowledgeFromType(resultTy);
184 
185  // Compute the knowledge based on the inferred type.
186  auto inferredKnowledge =
187  ValueKnowledge::getPessimisticValueState(op->getContext());
188  inferredKnowledge.dtype =
189  resultTy.cast<ShapedType>().getElementType();
190  inferredKnowledge.hasSizes = predictedShape.hasRank();
191  if (predictedShape.hasRank()) {
192  for (auto dim : predictedShape.getDims()) {
193  inferredKnowledge.sizes.push_back(dim);
194  }
195  }
196 
197  if (!replaceable)
198  continue;
199 
200  // Compute the new type based on the joined version.
201  auto newKnowledge =
202  ValueKnowledge::join(currentKnowledge, inferredKnowledge);
203  result.setType(newKnowledge.getType());
204  }
205  }
206  });
207 
208  // Insert UnrealizedConversionCasts to guarantee ReturnOp agress with
209  // the FuncOp type.
210  func.walk([&](ReturnOp op) {
211  FuncOp parent = dyn_cast<FuncOp>(op->getParentOp());
212  if (!parent)
213  return;
214 
215  rewriter.setInsertionPoint(op);
216  FunctionType funcTy = func.getType();
217  auto resultTys = funcTy.getResults();
218 
219  bool castAdded = false;
220  SmallVector<Value> castedValues;
221  for (auto it : llvm::zip(op->getOperands(), resultTys)) {
222  auto operand = std::get<0>(it);
223  auto currentTy = operand.getType();
224  auto castTy = std::get<1>(it);
225  if (currentTy == castTy) {
226  castedValues.push_back(operand);
227  continue;
228  }
229 
230  castedValues.push_back(
231  rewriter.create<tensor::CastOp>(op.getLoc(), castTy, operand)
232  .getResult());
233 
234  castAdded = true;
235  }
236 
237  if (castAdded) {
238  rewriter.replaceOpWithNewOp<ReturnOp>(op, castedValues);
239  }
240  });
241  }
242 };
243 } // end anonymous namespace
