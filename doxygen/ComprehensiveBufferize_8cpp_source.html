9 // Perform inplace bufferization within function boundaries.
10 // This is a specialized pass that supports inplace analysis for a fixed subset
11 // of ops that have well-defined inplace semantics.
12 // This pass caters to high-performance codegen where buffer reuse is deemed
13 // critical: the pass should fail if the bufferized form of the function needs
14 // to return any buffer.
15 // Generic control-flow and branching are unsupported.
16 // Composability with extensible set of ops is not a first-class concern.
17 //
18 // Bufferization occurs by:
19 // a. performing an inPlace analysis `inPlaceAnalysisFuncOpBody`
20 // which marks each operation within the function with the
21 // `kInPlaceResultsAttrName` attribute.
22 // b. traversing each operation in the function and rewriting it in
23 // buffer form and keeping a BlockAndValueMapping mapping of the
24 // rewrites. New allocations are introduced during this step.
25 // TODO: Allocation + depending op hoisting to outermost enclosing
26 // sequential scope.
27 // c. at the end of this bufferization, 3 cases may occur:
28 // i. inplaceable function arguments may be reused in place after the
29 // function itself has been bufferized. This is encoded by IR resembling:
30 //
31 // ```
32 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
33 // func @foo(%A: tensor<?xf32> {linalg.inplaceable = true})
34 // -> tensor<?xf32> {
35 // %0 = memref.buffer_cast %A : memref<?xf32, #map>
36 // // ... uses of %0
37 // %res = memref.tensor_load %0 : memref<?xf32, #map>
38 // return %res : tensor<?xf32>
39 // }
40 // ```
41 //
42 // this is the cue for the bufferization of the function foo (and calls
43 // to it) may bufferize to `func @foo(%A: memref<?xf32, some_layout>)`.
44 // To fully achieve bufferization, an additional analysis is needed to
45 // determine whether function argument/operand pairs bufferize to a
46 // single inplace buffer argument (i.e. functions may return tensors in
47 // arbitrary order that may not match argument numbers).
48 //
49 // ii. results that don't map to an inplaceable function argument are
50 // generally allocated. Since memref semantics wrt ownership of the
51 // underlying memory region are not well-defined, comprehensive
52 // bufferization chooses to perform allocations in a scoped fashion:
53 // returning memrefs is always considered illegal.
54 // Such scenarios are encoded by IR resembling:
55 //
56 // ```
57 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
58 // func @foo(%A: tensor<?xf32> {linalg.inplaceable = true})
59 // -> tensor<?xf32> {
60 // %0 = memref.buffer_cast %A : memref<?xf32, #map>
61 // %1 = memref.dim %0, %c0 : memref<?xf32, #map>
62 // %2 = memref.alloc(%1) : memref<?xf32>
63 // %3 = memref.cast %2 : memref<?xf32> to memref<?xf32, #map>
64 // // ... uses of %3
65 // memref.dealloc %2 : memref<?xf32, #map>
66 // %res = memref.tensor_load %3 : memref<?xf32, #map>
67 // return %res : tensor<?xf32>
68 // }
69 // ```
70 //
71 // this is the cue for the bufferization of the function foo (and calls
72 // to it) that it must bufferize to `func @foo(%A: memref<?xf32,
73 // some_layout>,
74 // %B: memref<?xf32, some_layout>)` (i.e. make a cloned
75 // allocation of the result tensor)
76 // To fully achieve bufferization, the alloc/dealloc pair must be lifted
77 // out of the function at each call site.
78 //
79 // iii. as an optimization over ii., it may be possible to reuse an argument
80 // and only want to return a slice.
81 // This may forego allocation by letting *all* callers decide whether to
82 // pass a new *aliasing* memref function argument (i.e. a subview).
83 // Without loss of generality, callers may agree to allocate a new buffer
84 // to avoid this aliasing. Such scenarios are encoded by IR resembling:
85 //
86 // ```
87 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
88 // func @foo(%arg0: tensor<?xf32> {linalg.inplaceable = true})
89 // -> tensor<4xf32> {
90 // %0 = memref.buffer_cast %arg0 : memref<?xf32, #map>
91 // %1 = memref.subview %0[0] [4] [1] : memref<?xf32, #map> to
92 // memref<4xf32, #map>
93 // // ... inplace computes into %1
94 // %3 = memref.tensor_load %1 : memref<4xf32, #map>
95 // return %3 : tensor<4xf32>
96 // }
97 // ```
98 //
99 // Note: In the future, it may be worthwhile to design special bufferization
100 // ops to encode the desired semantics at function boundaries for i., ii. and
101 // iii.
102 //
103 // Lastly, note that layout map chosen to bufferize is the most dynamic
104 // canonical strided layout of the proper rank. This ensures compatibility with
105 // expected layouts after transformations. Combinations of memref.cast +
106 // canonicalization are responsible for clean ups.

246 /// Mark whether OpResult can actually be bufferized inplace.
247 /// If `inPlace` is `InPlaceSpec::True`, the use-def chain analysis has
248 /// guaranteed that no subsequent write would occur to the bufferized
249 /// tensor value (i.e. the result can be bufferized inPlace).
250 static void setInPlaceOpResult(OpResult opResult,
251  InPlaceSpec inPlace = InPlaceSpec::True) {
252  if (!opResult)
253  return;
254 
255  Operation *op = opResult.getOwner();
256  auto attr =
257  op->getAttr(kInPlaceResultsAttrName).dyn_cast_or_null<ArrayAttr>();
258  SmallVector<StringRef> inPlaceVector =
259  attr ? SmallVector<StringRef>(
260  llvm::to_vector<4>(attr.getAsValueRange<StringAttr>()))
261  : SmallVector<StringRef>(op->getNumResults(),
262  stringify(InPlaceSpec::None));
263  LDBG("->set inPlace=" << stringify(inPlace) << ": " << *op
264  << " @idx=" << opResult.getResultNumber() << '\n');
265  inPlaceVector[opResult.getResultNumber()] = stringify(inPlace);
266  op->setAttr(kInPlaceResultsAttrName,
267  OpBuilder(op).getStrArrayAttr(inPlaceVector));
268 }

270 /// Get the InPlaceSpec attribute entry `kInPlaceResultsAttrName` for
271 /// `opResult`. If the result is `InPlaceSpec::True`, the use-def chain analysis
272 /// has guaranteed that no subsequent read of the tensor value occurs and the
273 /// result can be buferized inPlace.
274 /// If no InPlaceSpec attribute has been set for `opResult`, return
275 /// InPlaceSpec::None.
276 static InPlaceSpec getInPlace(OpResult opResult) {
277  if (!opResult)
278  return InPlaceSpec::None;
279 
280  Operation *op = opResult.getOwner();
281  auto attr =
282  op->getAttr(kInPlaceResultsAttrName).dyn_cast_or_null<ArrayAttr>();
283  if (!attr)
284  return InPlaceSpec::None;
285 
286  // Must return a proper value.
287  return *symbolize(*(attr.getAsValueRange<StringAttr>().begin() +
288  opResult.getResultNumber()));
289 }

291 /// Get inPlace information for `bbArg`.
292 /// FuncOp allow argument attributes, we use those to encode the information.
293 /// BlockArgument of other ops delegate to their owner's parent op.
294 static InPlaceSpec getInPlace(BlockArgument bbArg) {
295  if (auto funcOp = dyn_cast<FuncOp>(bbArg.getOwner()->getParentOp())) {
296  BoolAttr inplaceAttr = funcOp.getArgAttrOfType<BoolAttr>(
297  bbArg.getArgNumber(), LinalgDialect::kInplaceableAttrName);
298  if (!inplaceAttr)
299  return InPlaceSpec::None;
300  return inplaceAttr.getValue() ? InPlaceSpec::True : InPlaceSpec::False;
301  }
302  // Interestingly, scf::ForOp's and TiledLoop's bbArg can **always** be viewed
303  // inplace from the perspective of ops nested under:
304  // 1. Either the matching iter operand is not bufferized inplace and an
305  // alloc + optional copy makes the bbArg itself inplaceable.
306  // 2. Or the matching iter operand is bufferized inplace and bbArg just
307  // bufferizes to that too.
308  if (isa<scf::ForOp, TiledLoopOp>(bbArg.getOwner()->getParentOp()))
309  return InPlaceSpec::True;
310  // Unknown cases.
311  return InPlaceSpec::None;
312 }

512 /// Determine which OpResult will alias with `opOperand` if the op is bufferized
513 /// in place. This is a superset of `getInplaceableOpResult`.
514 /// Return None if the owner of `opOperand` does not have known
515 /// bufferization aliasing behavior, which indicates that the op must allocate
516 /// all of its tensor results.
517 /// TODO: in the future this may need to evolve towards a list of OpResult.
518 static Optional<OpResult> getAliasingOpResult(OpOperand &opOperand) {
519  if (!hasKnownBufferizationAliasingBehavior(opOperand.getOwner()))
520  return None;
521  return TypeSwitch<Operation *, OpResult>(opOperand.getOwner())
522  // These terminators legitimately have no result.
523  .Case<ReturnOp, linalg::YieldOp, scf::YieldOp>(
524  [&](auto op) { return OpResult(); })
525  // DimOp has no tensor result.
526  .Case<tensor::DimOp>([&](auto op) { return None; })
527  // ConstantOp is never inplaceable.
528  .Case([&](ConstantOp op) { return op->getResult(0); })
529  // ExtractSliceOp is different: its result is not inplaceable on op.source
530  // but when bufferized inplace, the result is an aliasing subregion of
531  // op.source.
532  .Case([&](ExtractSliceOp op) { return op->getResult(0); })
533  // All other ops, including scf::ForOp, return the result of
534  // `getInplaceableOpResult`.
535  .Default(
536  [&](Operation *op) { return getInplaceableOpResult(opOperand); });
537 }

539 /// Return true if `opOperand` bufferizes to a memory read.
540 static bool bufferizesToMemoryRead(OpOperand &opOperand) {
541  Optional<OpResult> maybeOpResult = getAliasingOpResult(opOperand);
542  // Unknown op that returns a tensor. The inplace analysis does not support
543  // it. Conservatively return true.
544  if (!maybeOpResult)
545  return true;
546  // ExtractSliceOp alone doesn't bufferize to a memory read, one of its uses
547  // may.
548  if (isa<ExtractSliceOp>(opOperand.getOwner()))
549  return false;
550  // scf::ForOp alone doesn't bufferize to a memory read, one of the uses of its
551  // matching bbArg may.
552  if (auto forOp = dyn_cast<scf::ForOp>(opOperand.getOwner())) {
553  for (OpOperand &use :
554  forOp.getRegionIterArgForOpOperand(opOperand).getUses())
555  if (bufferizesToMemoryRead(use))
556  return true;
557  return false;
558  }
559  // TiledLoop alone doesn't bufferize to a memory read, one of the uses of its
560  // matching bbArg may.
561  if (auto tiledLoopOp = dyn_cast<TiledLoopOp>(opOperand.getOwner())) {
562  for (OpOperand &use : tiledLoopOp.getTiedBlockArgument(opOperand).getUses())
563  if (bufferizesToMemoryRead(use))
564  return true;
565  return false;
566  }
567  // CallOpInterface alone doesn't bufferize to a memory read, one of the uses
568  // of the matching bbArg may. It is the responsibility of the caller to
569  // inspect bbArgs. In the absence of a BufferizationAliasInfo, we need to be
570  // conservative.
571  if (auto callOp = dyn_cast<CallOpInterface>(opOperand.getOwner()))
572  return true;
573  if (auto linalgOp = dyn_cast<LinalgOp>(opOperand.getOwner()))
574  return linalgOp.isInputTensor(&opOperand) ||
575  linalgOp.isInitTensor(&opOperand);
576  // All other cases are considered to bufferize to memory reads.
577  // In particular, terminators are often the last use and need to be considered
578  // as reads to return the proper value and avoid WAW clobbers.
579  return true;
580 }

582 /// Return true if `opOperand` bufferizes to a memory write.
583 /// If inPlaceSpec is different from InPlaceSpec::None, additionally require the
584 /// write to match the inplace specification.
585 static bool
586 bufferizesToMemoryWrite(OpOperand &opOperand,
587  InPlaceSpec inPlaceSpec = InPlaceSpec::None) {
588  // These terminators are not writes.
589  if (isa<ReturnOp, linalg::YieldOp, scf::YieldOp>(opOperand.getOwner()))
590  return false;
591  // ExtractSliceOp alone doesn't bufferize to a memory write, one of its uses
592  // may.
593  if (isa<ExtractSliceOp>(opOperand.getOwner()))
594  return false;
595  // CallOpInterface alone doesn't bufferize to a memory write, one of the uses
596  // of the matching bbArg may. It is the responsibility of the caller to
597  // inspect bbArgs. In the absence of a BufferizationAliasInfo, we need to be
598  // conservative.
599  if (auto callOp = dyn_cast<CallOpInterface>(opOperand.getOwner()))
600  return true;
601  Optional<OpResult> maybeOpResult = getAliasingOpResult(opOperand);
602  // Unknown op that returns a tensor. The inplace analysis does not support
603  // it. Conservatively return true.
604  if (!maybeOpResult)
605  return true;
606  // Supported op without a matching result for opOperand (e.g. ReturnOp).
607  // This does not bufferize to a write.
608  if (!*maybeOpResult)
609  return false;
610  // If we have a matching OpResult, this is a write.
611  // Additionally allow to restrict to only inPlace write, if so specified.
612  return inPlaceSpec == InPlaceSpec::None ||
613  getInPlace(*maybeOpResult) == inPlaceSpec;
614 }

907 /// Return true if it is possible to find an inplace write W among the uses of
908 /// aliasInfo[result], and a read R among the uses of aliasInfo[result],
909 /// such that W and R interfere.
910 /// Such a (W, R) pair is an interference to the inplace bufferization of
911 /// rootWrite when:
912 /// 1. R is not known properly dominate W (i.e. the effects of the write may
913 /// be visible from R).
914 /// 2. one cannot find an intermediate clobbering write `C` to W, such that
915 /// C interleaved between W and R (i.e. W -> C -> R where -> denotes
916 /// dominance).
917 bool BufferizationAliasInfo::wouldCreateReadAfterWriteInterference(
918  OpResult result, const DominanceInfo &domInfo) const {
919  Optional<OpOperand *> maybeAliasingOperand = getAliasingOpOperand(result);
920  if (!maybeAliasingOperand)
921  return false;
922 
923  Operation *opToBufferize = result.getDefiningOp();
924  Value root = (*maybeAliasingOperand)->get();
925  LDBG("----Start wouldCreateReadAfterWriteInterference\n");
926  LDBG("--------rootValue: " << root << "\n");
927 
928  // Collect:
929  // 1. all the inplace write uses of some alias of `root`.
930  // 2. all the write uses that belong to `opToBufferize`.
931  // opToBufferize is not yet inplace, we want to determine if it can be inplace
932  // so we also consider all its write uses, not just the inplace ones.
933  DenseSet<OpOperand *> usesWrite;
934  for (Value vWrite : getAliasInfoRef(root)) {
935  for (auto &uWrite : vWrite.getUses()) {
936  if (!bufferizesToMemoryWrite(uWrite))
937  continue;
938  if (uWrite.getOwner() == opToBufferize ||
939  bufferizesToMemoryWrite(uWrite, InPlaceSpec::True))
940  usesWrite.insert(&uWrite);
941  }
942  }
943  for (Value vWrite : getAliasInfoRef(result))
944  for (auto &uWrite : vWrite.getUses())
945  if (bufferizesToMemoryWrite(uWrite, InPlaceSpec::True))
946  usesWrite.insert(&uWrite);
947 
948  // Collect all the reads of some alias of `root`.
949  // opToBufferize is not yet inplace, we want to determine if it can be inplace
950  // so we also consider all read uses of its result.
951  DenseSet<OpOperand *> usesRead;
952  auto &aliasListRead = getAliasInfoRef(root);
953  for (Value vRead : aliasListRead)
954  for (auto &uRead : vRead.getUses())
955  if (bufferizesToMemoryRead(uRead))
956  usesRead.insert(&uRead);
957  for (Value vRead : getAliasInfoRef(result))
958  for (auto &uRead : vRead.getUses())
959  if (bufferizesToMemoryRead(uRead))
960  usesRead.insert(&uRead);
961 
962  for (OpOperand *uRead : usesRead) {
963  Operation *aliasingReadOp = uRead->getOwner();
964  LDBG("----++++aliasRead #" << uRead->getOperandNumber()
965  << " in: " << *aliasingReadOp << '\n');
966  for (OpOperand *uWrite : usesWrite) {
967  // Don't consider self-use of the same operand for interference.
968  // Multiple different uses within the same op is fair game though.
969  if (uWrite == uRead)
970  continue;
971 
972  Operation *aliasingWriteOp = uWrite->getOwner();
973  LDBG("---- aliasWrite #" << uWrite->getOperandNumber()
974  << " in: " << *aliasingWriteOp << '\n');
975  // If the candidate write is the one that produces the read value (in the
976  // SSA def-use sense), this is not considered an interference.
977  if (getInplaceableOpResult(*uWrite) == uRead->get())
978  continue;
979  // If aliasingReadOp properly dominates aliasingWriteOp, the read cannot
980  // be affected by the write: there is no interference.
981  if (domInfo.properlyDominates(aliasingReadOp, aliasingWriteOp))
982  continue;
983  // At this point, aliasingWriteOp properly dominates aliasingReadOp or
984  // there is no clear dominance and we need to be conservative.
985  LDBG("---->found RaW interference\n");
986  LDBG(" Interfering read (op #" << uRead->getOperandNumber()
987  << "): " << *aliasingReadOp << '\n');
988  LDBG(" Interfering write (op #" << uWrite->getOperandNumber()
989  << "): " << *aliasingWriteOp << '\n');
990  LDBG("---->opportunity to clobber RaW interference\n");
991  if (isClobberedWriteBeforeRead(opToBufferize, *uRead, *uWrite, domInfo)) {
992  LDBG("---->clobbered! -> skip\n");
993  continue;
994  }
995 
996  LDBG("---->not clobbered -> found an interference\n");
997  return true;
998  }
999  }
1000  LDBG("----No interference found\n");
1001  return false;
1002 }

1070 /// Union all the aliasing sets of all aliases of v1 and v2.
1071 bool BufferizationAliasInfo::mergeAliases(Value v1, Value v2) {
1072  // Avoid invalidation of iterators by pre unioning the aliases for v1 and v2.
1073  bool changed = set_union(getAliasInfoRef(v1), getAliasInfoRef(v2)) ||
1074  set_union(getAliasInfoRef(v2), getAliasInfoRef(v1));
1075  for (auto v : getAliasInfoRef(v1))
1076  if (v != v1)
1077  changed |= set_union(getAliasInfoRef(v), getAliasInfoRef(v2));
1078  for (auto v : getAliasInfoRef(v2))
1079  if (v != v2)
1080  changed |= set_union(getAliasInfoRef(v), getAliasInfoRef(v1));
1081  return changed;
1082 }

1182  // ... = insert_slice %W into %R[%offset_sizes_and_strides_1]
1183  if (aliasingRead.get() == insertSliceOp.dest() &&
1184  // TODO: This is currently too restrictive and misses clobberings.

1186  // should be that the `aliasingWrite` is contained within
1187  // `insertSliceOp.source()`.

1191  LDBG("---->clobbering matching extract_slice/insert_slice\n");

1197  // %X = inplace_write(%1)

1200  // // aliasingWrite %Y in insert_slice

1202  if (aliasingReadOp == aliasingWriteOp) {
1203  assert(aliasingRead.get() == insertSliceOp.source() &&

1212  }

1218  // equivalence). We will need to compute container-containee relationship.

1378 /// Helper function for LinalgOp bufferization.
1379 /// Examines each result and determines whether it bufferizes inplace on an
1380 /// operand.
1381 /// If the opResult bufferizes inplace, just reuse the existing buffer.
1382 /// Otherwise allocate a new buffer to hold the result.
1383 /// When allocating a new buffer, analyze whether `op` want to read form that
1384 /// buffer. In such a case, insert a copy to ensure the newly allocated buffer
1385 /// is properly initialiazed.
1386 static void allocateBuffersForResults(OpBuilder &b, Location loc, LinalgOp op,
1387  SmallVectorImpl<Value> &resultBuffers,
1388  BlockAndValueMapping &bvm,
1389  BufferizationAliasInfo &aliasInfo) {
1390  // Take a guard before anything else.
1391  OpBuilder::InsertionGuard g(b);
1392 
1393  // TODO: provide the proper interface to iterate on OpResults and get the
1394  // matching OpOperands.
1395  for (OpOperand *opOperand : op.getOutputOperands()) {
1396  Value output = opOperand->get();
1397  assert(output.getType().isa<TensorType>() && "expected tensor type");
1398 
1399  // If output tensor is marked inPlace, just use the buffer.
1400  // The following uses internal knowledge of the position of inplaceable
1401  // operand / results.
1402  OpResult opResult = getInplaceableOpResult(*opOperand);
1403  if (getInPlace(opResult) == InPlaceSpec::True) {
1404  Value v = lookup(bvm, output);
1405  assert(v && "missing buffer");
1406  resultBuffers.push_back(v);
1407  continue;
1408  }
1409 
1410  // Otherwise, `op` is not inplaceable and we need to allocate its result.
1411  Value dimTensor = bvm.lookupOrDefault(output);
1412  Value alloc =
1413  createNewAllocDeallocPairForShapedValue(b, loc, dimTensor, aliasInfo);
1414  b.setInsertionPointAfter(alloc.getDefiningOp());
1415  resultBuffers.push_back(alloc);
1416 
1417  // Additionally, if the output buffer is used, clone its value for now.
1418  if (op.payloadUsesValueFromOperand(opOperand)) {
1419  Value v = lookup(bvm, output);
1420  b.create<CopyOp>(loc, v, alloc);
1421  }
1422  }
1423 
1424  if (op->getNumResults())
1425  map(bvm, op->getResults(), resultBuffers);
1426 }

1436  // basis.

1470 /// In a first approximation, all the function arguments of a FuncOp are marked
1471 /// inplaceable. For now, it is the responsibility of the `callOp` bufferization
1472 /// to allow FuncOp that are inplaceable to write inPlace.
1473 static LogicalResult
1474 bufferize(OpBuilder &b, CallOpInterface callOp, BlockAndValueMapping &bvm,
1475  BufferizationAliasInfo &aliasInfo,
1476  DenseMap<FuncOp, FunctionType> &bufferizedFunctionTypes) {
1477  FuncOp funcOp = getCalledFunction(callOp);
1478  assert(isa<CallOp>(callOp.getOperation()) && funcOp &&
1479  "expected Callop to a FuncOp");
1480 
1481  // If nothing to do then we are done.
1482  if (!llvm::any_of(funcOp.getType().getInputs(), isaTensor) &&
1483  !llvm::any_of(funcOp.getType().getResults(), isaTensor))
1484  return success();
1485 
1486  // Take a guard before anything else.
1487  OpBuilder::InsertionGuard g(b);
1488  b.setInsertionPoint(callOp);
1489 
1490  // 1. Filter return types:
1491  // - if the callee is bodiless / external, we cannot inspect it and we
1492  // cannot assume anything. We can just assert that it does not return a
1493  // tensor as this would have to bufferize to "return a memref", whose
1494  // semantics is ill-defined.
1495  // - if the callee has a body, we perform inter-procedural equivalence
1496  // analysis. When successful, a result folds onto an operand. When
1497  // unsuccessful, additional work is needed to either:
1498  // * hoist a result into an inplaceable operand or
1499  // * devise a better representation to truly return a buffer.
1500  SmallVector<Type> resultTypes;
1501  SmallVector<Value> hoistedArguments;
1502  if (funcOp.body().empty()) {
1503  if (llvm::any_of(funcOp.getType().getResults(), isaTensor))
1504  return callOp->emitError()
1505  << "cannot bufferize bodiless function that returns a tensor";
1506  } else {
1507  ReturnOp returnOp = getAssumedUniqueReturnOp(funcOp);
1508  assert(returnOp && "expected func with single return op");
1509 
1510  // For each FuncOp result, keep track of which inplace argument it reuses.
1511  for (OpOperand &returnOperand : returnOp->getOpOperands()) {
1512  Type returnType = returnOperand.get().getType();
1513  if (!isaTensor(returnType)) {
1514  resultTypes.push_back(returnType);
1515  continue;
1516  }
1517 
1518  // If return operand is equivalent to some bbArg, no need to return it.
1519  Value returnVal = returnOperand.get();
1520  if (BlockArgument bbArg =
1521  getEquivalentEnclosingFuncBBArg(returnVal, aliasInfo)) {
1522  Value oldRes = callOp->getResult(returnOperand.getOperandNumber());
1523  int64_t idx = bbArg.getArgNumber();
1524  Value buffer = lookup(bvm, callOp->getOperand(idx));
1525  assert(buffer && "expected bufferized value");
1526  // Add CallOp operand/result equivalence: this is interprocedural info.
1527  aliasInfo.insertNewBufferEquivalence(oldRes, buffer);
1528  map(bvm, oldRes, buffer);
1529  // Add a TensorLoadOp to kill all uses of the CallOp return.
1530  // Replace all uses of the CallOp results so we can erase the CallOp.
1531  // This TensorLoadOp must fold/DCE away or bufferization should be
1532  // considered failed.
1533  Value tensorLoad =
1534  b.create<memref::TensorLoadOp>(callOp.getLoc(), buffer);
1535  oldRes.replaceAllUsesWith(tensorLoad);
1536  // Add new op equivalence info.
1537  aliasInfo.insertNewBufferEquivalence(tensorLoad, buffer);
1538  map(bvm, tensorLoad, buffer);
1539  continue;
1540  }
1541 
1542  // TODO: Need to hoist above function boundary.
1543  if (Operation *allocOp = getEquivalentAlloc(returnVal, aliasInfo)) {
1544  hoistedArguments.push_back(allocOp->getResult(0));
1545  continue;
1546  }
1547 
1548  // Other cases legitimately need to return a tensor, this is currently not
1549  // supported. For instance, if hoisting across function boundary has
1550  // failed, it may be due to e.g. data-dependent sizes. In such a case, we
1551  // would we need a better type than memref.
1552  resultTypes.push_back(returnType);
1553 
1554  int64_t returnIdx = returnOperand.getOperandNumber();
1555  return returnOp->emitError()
1556  << "buffer result #" << returnIdx << " not produced by an alloc\n";
1557  }
1558  }
1559 
1560  // 2. Compute bufferized FunctionType.
1561  SmallVector<Type> argumentTypes{callOp->getOperandTypes()};
1562  ValueRange hoistedArgs{hoistedArguments};
1563  llvm::append_range(argumentTypes, hoistedArgs.getTypes());
1564  // Get the bufferized FunctionType for funcOp or construct it if not yet
1565  // available.
1566  FunctionType bufferizedFuncType = getOrCreateBufferizedFunctionType(
1567  funcOp, argumentTypes, resultTypes, bufferizedFunctionTypes);
1568 
1569  // 3. Rewrite tensor operands as memrefs based on `bufferizedFuncType`.
1570  SmallVector<Value> newOperands;
1571  newOperands.reserve(callOp->getNumOperands());
1572  for (OpOperand &opOperand : callOp->getOpOperands()) {
1573  Value tensorOperand = opOperand.get();
1574  // Non-tensor operands are just copied.
1575  if (!tensorOperand.getType().isa<TensorType>()) {
1576  newOperands.push_back(tensorOperand);
1577  continue;
1578  }
1579 
1580  // Tensor operands are guaranteed to have been buferized.
1581  int64_t idx = opOperand.getOperandNumber();
1582  Value buffer = lookup(bvm, tensorOperand);
1583  assert(buffer && "expected bufferized value");
1584 
1585  // Caller / callee type mistmatch is handled with a CastOp.
1586  auto memRefType = bufferizedFuncType.getInput(idx);
1587  // Since we don't yet have a clear layout story, buffer_cast may
1588  // conservatively turn tensors into more dynamic memref than necessary.
1589  // If the memref type of the callee fails, introduce an extra memref.cast
1590  // that will either canonicalize away or fail compilation until we can do
1591  // something better.
1592  if (buffer.getType() != memRefType) {
1593  Value castBuffer =
1594  b.create<memref::CastOp>(callOp.getLoc(), memRefType, buffer);
1595  // Add new op equivalence info.
1596  aliasInfo.insertNewBufferEquivalence(castBuffer, buffer);
1597  map(bvm, tensorOperand, castBuffer);
1598  buffer = castBuffer;
1599  }
1600  newOperands.push_back(buffer);
1601  }
1602 
1603  // 4. Create the new CallOp.
1604  Operation *newCallOp = b.create<CallOp>(callOp.getLoc(), funcOp.sym_name(),
1605  resultTypes, newOperands);
1606  newCallOp->setAttrs(callOp->getAttrs());
1607  return success();
1608 }

1797  int numControlOperands = tiledLoopOp.getNumControlOperands();

1800  // Keep separate iterators to increment without further leaking impl. details.
1801  // Start with outputs to avoid interference from new input buffers.
1802  int numNewOutputBuffers = 0;
1803  int resultIndex = 0;
1804  int oldOutputBBArgIndex = numLoops + oldInputs.size();
1805  int nextOutputBBArgIndex = numLoops + oldInputs.size() + oldOutputs.size();
1806  int nextOutputOperandIndex =
1807  numControlOperands + oldInputs.size() + oldOutputs.size();

1809  if (!oldOutputTensor.getType().isa<TensorType>()) {
1810  // Skip and increment the old bbarg index only.
1811  ++oldOutputBBArgIndex;

1813  // TODO: better interface to avoid leaking such impl details.
1814  continue;
1815  }
1816 
1817  assert(oldOutputTensor.getType().isa<RankedTensorType>() &&
1818  "bufferizable output must be a ranked tensor");

1821  const OpResult &opResult = tiledLoopOp->getResult(resultIndex);
1822  OpOperand &yieldOperand = yieldOp->getOpOperand(resultIndex);

1829  // unitialized and we do not need to copy.
1830  // TODO: "matching bbArg does not bufferize to a read" is a more general
1831  // check.
1832  if (!oldOutputTensor.getDefiningOp<linalg::InitTensorOp>()) {
1833  b.setInsertionPointAfter(alloc.getDefiningOp());

1849  aliasInfo.createAliasInfoEntry(newBufferBBArg);
1850  aliasInfo.insertNewBufferEquivalence(oldTensorBBArg, newBufferBBArg);
1851  map(bvm, oldTensorBBArg, newBufferBBArg);
1852

1854  // later.
1855  yieldOperand.set(oldTensorBBArg);
1856 
1857  // Increment indices.
1858  ++numNewOutputBuffers;
1859  ++resultIndex;
1860  ++oldOutputBBArgIndex;
1861  ++nextOutputBBArgIndex;
1862  ++nextOutputOperandIndex;
1863  }
1864 
1865  // Add buffers for inputs and the corresponding block arguments.
1866  // Keep separate iterators to increment without further leaking impl. details.
1867  int numNewInputBuffers = 0;
1868  int oldInputBBArgIndex = numLoops;
1869  int nextInputBBArgIndex = numLoops + oldInputs.size();
1870  int nextInputOperandIndex = numControlOperands + oldInputs.size();
1871  for (Value oldInputTensor : oldInputs) {
1872  if (!oldInputTensor.getType().isa<TensorType>()) {
1873  // Skip and increment the old bbarg index only.
1874  ++oldInputBBArgIndex;
1875  continue;
1876  }
1877 
1878  Value inputBuffer = lookup(bvm, oldInputTensor);
1879  assert(inputBuffer && " missing buffer for operand");
1880 
1881  // Insert new operand and bbArg.
1882  tiledLoopOp->insertOperands(nextInputOperandIndex, inputBuffer);
1883  BlockArgument newBufferBBArg =
1884  body->insertArgument(nextInputBBArgIndex, inputBuffer.getType());
1885  BlockArgument oldTensorBBArg = body->getArgument(oldInputBBArgIndex);
1886 
1887  // Insert mapping and aliasing info.
1888  aliasInfo.createAliasInfoEntry(newBufferBBArg);
1889  aliasInfo.insertNewBufferEquivalence(oldTensorBBArg, newBufferBBArg);
1890  map(bvm, oldTensorBBArg, newBufferBBArg);
1891 
1892  // Increment indices.
1893  ++numNewInputBuffers;
1894  ++oldInputBBArgIndex;
1895  ++nextInputBBArgIndex;
1896  ++nextInputOperandIndex;
1897  }
1898

1900  // TODO: Helper method to avoid leaking impl details.
1901  tiledLoopOp->setAttr(
1902  TiledLoopOp::getOperandSegmentSizeAttr(),
1903  b.getI32VectorAttr(
1904  {numLoops, numLoops, numLoops,
1905  static_cast<int>(oldInputs.size()) + numNewInputBuffers,
1906  static_cast<int>(oldOutputs.size()) + numNewOutputBuffers}));

2149 ///
2150 /// Rationale for bufferizing `%1 = tensor.extract_slice %0[...]` inplace.
2151 /// ===========================================================
2152 ///
2153 /// When bufferized out of place, a ExtractSlice lowers to alloc + copy. This
2154 /// cannot change the flow of information for either the source or the
2155 /// result buffers.
2156 ///
2157 /// When bufferized inplace, a ExtractSliceOp does not by itself create any read
2158 /// or write from memory. Instead, it has the effect of merging the alias sets
2159 /// of the source and the result buffers.
2160 ///
2161 /// An analysis is required to ensure inplace bufferization would not result in
2162 /// RaW dependence violations.
2163 static LogicalResult
2164 bufferizableInPlaceAnalysis(ExtractSliceOp extractSliceOp,
2165  BufferizationAliasInfo &aliasInfo,
2166  const DominanceInfo &domInfo) {
2167  LDBG('\n');
2168  LDBG("Inplace analysis for extract_slice: " << *extractSliceOp << '\n');
2169 
2170  // If `extractSliceOp` were to be bufferized inplace, it cannot end up
2171  // aliasing a write into a non-writeable buffer.
2172  bool wouldCreateAliasingWriteToNonWriteableBuffer =
2173  aliasInfo.aliasesInPlaceWrite(extractSliceOp.result()) &&
2174  aliasInfo.aliasesNonWriteableBuffer(extractSliceOp->getOpOperand(0));
2175 
2176  if (wouldCreateAliasingWriteToNonWriteableBuffer)
2177  LDBG("->the corresponding buffer is not writeable\n");
2178  else
2179  LDBG("->bufferizes to writeable inplace buffer\n");
2180 
2181  // In any of extractSliceOp.result's aliases, can we find 2 such that we hit
2182  // an interfering write?
2183  OpResult r = extractSliceOp->getResult(0);
2184  OpOperand &s = extractSliceOp->getOpOperand(0);
2185  bool foundInterference =
2186  wouldCreateAliasingWriteToNonWriteableBuffer ||
2187  aliasInfo.wouldCreateReadAfterWriteInterference(r, domInfo);
2188  if (foundInterference)
2189  aliasInfo.bufferizeOutOfPlace(r);
2190  else
2191  aliasInfo.bufferizeInPlace(r, s);
2192 
2193  LDBG("Done inplace analysis for extract_slice\n");
2194 
2195  return success();
2196 }

2427 /// buffer form (using the canonical memref layout for now), according to the

2429 /// This relies on a buffer equivalence analysis of each return operand. When a
2430 /// result buffer is equivalent to:
2431 /// 1. a BlockArgument of `funcOp`, it can be dropped from the return values
2432 /// and becomes inplaceable at all callers. This assumes all CallOp perform
2433 /// the necessary work to clone operands so as to make them inplaceable.

2436 /// candidate for hoisting and creating a new inplace operand at all caller

2457  // result tensors.

2579 getFuncOpsOrderedByCalls(ModuleOp moduleOp,
2580  SmallVectorImpl<FuncOp> &orderedFuncOps,
2581  DenseMap<FuncOp, DenseSet<Operation *>> &callerMap) {

2743 
2744  SmallVector<FuncOp> orderedFuncOps;
2745  DenseMap<FuncOp, DenseSet<Operation *>> callerMap;
2746  DenseMap<FuncOp, FunctionType> bufferizedFunctionTypes;
2747  if (failed(getFuncOpsOrderedByCalls(moduleOp, orderedFuncOps, callerMap)))
2748  return signalPassFailure();

2751  DominanceInfo domInfo(moduleOp);
2752  BufferizationAliasInfo aliasInfo(moduleOp);
2753  // Interestingly, all function args that are not visible outside of a module
2754  // can be fully bufferized inplace by guaranteeing the CallOp is bufferized
2755  // inplace. Therefore, we just bufferize funcOp as if none of its results were
2756  // inplaceable, detect which operands are cloned internally and decide what to
2757  // do at call sites.
2758  for (FuncOp funcOp : orderedFuncOps) {
2759  // No body => no analysis.
2760  if (funcOp.body().empty())
2761  continue;
2762 
2763  // In a first approximation:
2764  // =========================
2765  // If the function is called, we can allocate on the caller side which lets
2766  // us force inplace arguments at function boundaries.
2767  // TODO: do not rely on this behavior.
2768  if (callerMap.find(funcOp) != callerMap.end())
2769  for (BlockArgument bbArg : funcOp.getArguments())
2770  if (bbArg.getType().isa<TensorType>())
2771  setInPlaceFuncArgument(bbArg);
2772 
2773  // If the analysis fails, just return.
2774  if (failed(inPlaceAnalysisFuncOpBody(funcOp, aliasInfo, domInfo))) {
2775  signalPassFailure();
2776  return;
2777  }
2778 
2779  // Bufferization phase.
2780  if (!testAnalysisOnly) {
2781  BlockAndValueMapping tensorToBufferMap;
2782  if (failed(bufferizeFuncOpInternals(funcOp, tensorToBufferMap, aliasInfo,
2783  bufferizedFunctionTypes,
2784  globalCreator))) {
2785  signalPassFailure();
2786  return;
2787  }
2788  }
2789  }
2790  // Don't drop the attributes if we only want to report the analysis.
2791  if (testAnalysisOnly)
2792  return;
2793 
2794  for (FuncOp funcOp : orderedFuncOps) {
2795  // Note: It would be good to apply cleanups here but we cannot as aliasInfo
2796  // would be invalidated.
2797  if (failed(bufferizeFuncOpBoundary(funcOp, aliasInfo,
2798  bufferizedFunctionTypes))) {
2799  signalPassFailure();
2800  return;
2801  }
2802  }
2803

2805  // according to the kBufferLayoutAttrName.
2806  layoutPostProcessing(moduleOp);
2807

2809  moduleOp.walk(

2813  removeBufferizationFuncArguments(bbArg);
2814  });

2821 }
