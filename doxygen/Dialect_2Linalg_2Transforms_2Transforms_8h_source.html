153 /// Fuse a sequence of linalg operations (`ops`) using tile-and-fuse. This
154 /// proceeds as follows:
155 /// - Find outer parallel loops in these ops that can be fused.
156 /// - Tile fusable outer parallel loops of the last operation in the sequence.
157 /// - Fuse the remaining operations with the tiled operation
158 ///
159 /// For example, consider the sequence of matmul below
160 ///
161 /// linalg.matmul ins(%arg0, %arg1 : memref<256x32xf32>, memref<32x32xf32>)
162 /// outs(%arg2 : memref<256x32xf32>)
163 /// linalg.matmul ins(%arg2, %arg3 : memref<256x32xf32>, memref<32x32xf32>)
164 /// outs(%arg4 : memref<256x32xf32>)
165 ///
166 /// It is legal to fuse the RAW dependence (through %arg2) by only fusing the
167 /// matmuls row-wise. For example, the fused computation for the above is shown
168 /// below. The outer `scf.parallel` loop is the "fused" loop obtained by tiling
169 /// along the rows of the matrix. The entire rows of the first matmul operation
170 /// need to be computed before they can be used for the second matmul. The
171 /// second matmul is further tiled (similar to normal tiling).
172 ///
173 /// #map0 = affine_map<(d0, d1)[s0] -> (d0 * 32 + s0 + d1)>
174 /// #map1 = affine_map<(d0, d1) -> (d0 * 32 + d1)>
175 /// scf.parallel (%arg5) = (%c0) to (%c256) step (%c16) {
176 /// %0 = subview %arg2[%arg5, 0] [16, 32] [1, 1]
177 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
178 /// %1 = subview %arg4[%arg5, 0] [16, 32] [1, 1]
179 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
180 /// %2 = subview %arg0[%arg5, 0] [16, 32] [1, 1]
181 /// : memref<256x32xf32> to memref<16x32xf32, #map0>
182 /// %3 = subview %arg1[0, 0] [32, 32] [1, 1]
183 /// : memref<32x32xf32> to memref<32x32xf32, #map1>
184 /// %4 = subview %arg3[0, 0] [32, 32] [1, 1]
185 /// : memref<32x32xf32> to memref<32x32xf32, #map1>
186 /// linalg.matmul
187 /// ins(%2, %3 : memref<16x32xf32, #map0>, memref<32x32xf32, #map1>)
188 /// outs(%0 : memref<16x32xf32, #map0>)
189 /// linalg.matmul
190 /// ins(%0, %4 : memref<16x4xf32, #map0>, memref<4x8xf32, #map0>)
191 /// outs(%1 : memref<16x8xf32, #map0>)
192 /// }
193 ///
194 /// `tilingOptions` are used to tile the corresponding operation in `ops` (the
195 /// size of the former should be same as size of the latter. Based on how
196 /// tile+fuse is implemented, the fused loops are generated based on the last
197 /// operation in the sequence. For example, the tile sizes for the fused loops
198 /// is obtained from `tilingOptions.back()`. The following tiling options are
199 /// handled differently in tile+fuse (compared to tile only)
200 /// - Interchange of the tiling loops is not supported right now.
201 /// - Only the fused loops are distributed.
202 struct TiledAndFusedLinalgOps {
203  /// Operation obtained by tiling the last operation in sequence of `ops`
204  /// passed to `tileAndFuseLinalgOps`.
205  LinalgOp op;
206  /// The dimension of the loops that are fused.
207  std::set<unsigned> fusedLoopDims;
208  /// The generated fused operations (created within the fused loops).
209  SmallVector<LinalgOp, 1> fusedProducers;
210  /// The fused loop generated.
211  SmallVector<Operation *, 4> fusedLoops;
212 };

1041 /// Converts Convolution op into vector contraction.
1042 ///
1043 /// Conversion expects ConvOp to have dimensions marked in the *mask* as
1044 /// false of size 1. This ensures that the ConvOp can be lowered to vector
1045 /// contraction of dimensions marked in the *mask* as true.
1046 ///
1047 /// A good example for vectorization is ConvNHWCOp which is 2D Conv op
1048 /// with channels as the last dimension. Let's vectorize last 3 dimensions.
1049 /// The initial op definition looks like this:
1050 /// ```
1051 /// linalg.conv_2d_nhwc %arg0, %arg1, %arg2 :
1052 /// (memref<1x3x3x3xf32>, memref<1x3x3x3xf32>, memref<?x?x?x?xf32>)
1053 /// ```
1054 /// This op can be expressed as a dot product between %arg0 (input) and
1055 /// %arg1 (kernel) which is written into first entry of %arg2 (output). This is
1056 /// the ConvOp this pass expects and converts into:
1057 /// ```
1058 /// #map0 = affine_map<(d0, d1, d2) -> (d0, d1, d2)>
1059 /// #map1 = affine_map<(d0, d1, d2) -> ()>
1060 /// .....
1061 /// %0 = vector.transfer_read %arg0[%c0, %c0, %c0, %c0], %c0_f32
1062 /// : memref<1x3x3x3xf32>, vector<3x3x3xf32>
1063 /// %1 = vector.transfer_read %arg1[%c0, %c0, %c0, %c0], %c0_f32
1064 /// : memref<1x3x3x3xf32>, vector<3x3x3xf32>
1065 /// %2 = vector.contract {indexing_maps = [#map0, #map0, #map1],
1066 /// iterator_types = ["reduction", "reduction", "reduction"]} %0, %1,
1067 /// %c0_f32 : vector<3x3x3xf32>, vector<3x3x3xf32> into f32
1068 /// store %2, %arg2[%c0, %c0, %c0, %c0] : memref<?x?x?x?xf32>
1069 /// ```
1070 /// where first 2 operations read input and kernel memory buffers into vectors.
1071 /// Subsequently, they are contracted together and the result is written to
1072 /// the first entry of the output buffer.
